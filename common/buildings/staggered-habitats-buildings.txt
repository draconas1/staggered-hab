#special buildings for colonised habitats
building_stag_hab_space_elevator = {
	base_buildtime = 500
	planet_unique = yes
	
	cost = {
		minerals = 750		
	}
	
	potential = {
		planet = {
			is_planet_class = pc_habitat
		}
		owner = { has_ascension_perk = ap_voidborn }
	}	
	
	allow = {
		planet = { 
			has_building = "building_hab_capital"
		}
		custom_tooltip = {
			fail_text = "requires_sync_orbit"
			tile = {
				has_resource = {
					type = sr_stag_hab_sync_orbit
					amount > 0
				}
			}
		}
	}
	
	produced_resources = {
		minerals = 4
	}
	
	planet_modifier  = {
		tile_resource_minerals_mult = 0.20
	}
	
	adjacency_bonus = {
		tile_building_resource_minerals_add = 3
		tile_building_resource_food_add = 2
		tile_building_resource_energy_add = 2
	}
	
	required_resources = {
		energy = 5
	}
	
	prerequisites = {
		"tech_mineral_processing_2"
	}
}
